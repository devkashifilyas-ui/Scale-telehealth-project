// Configurable retry schedule per plan const RETRY_SCHEDULE: number[] = [24, 72, 168]; // hours async function processDunningAttempt( sub: Subscription, attempt: number ): Promise<DunningResult> { const result = await stripe.retryPayment(sub.stripeSubId); if (result.success) { await transitionState(sub.id, 'RECOVERED'); await ghl.triggerWorkflow(sub.patientId, 'payment-recovered'); return { recovered: true }; } if (attempt >= RETRY_SCHEDULE.length) { await transitionState(sub.id, 'CANCELLED'); return { cancelled: true }; } // Schedule next retry via SQS delay queue await queue.enqueue('RETRY_PAYMENT', { subscriptionId: sub.id, attempt: attempt + 1, delayHours: RETRY_SCHEDULE[attempt] }); }
